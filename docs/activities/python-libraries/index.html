import pyfiglet

# Get the list of fonts and sort it.

font_names = sorted(pyfiglet.FigletFont.getFonts())

print(font_names)

Documentation: https://docs.python.org/3/library/random.html

Code sample:

import random

# Generate random integers (int).

for i in range(5):
    print(random.randint(5, 10))

# Possible output:
#     7
#     10
#     9
#     7
#     8
        
# Generate random floating point numbers (float).

for i in range(5):
    print(random.random())

# Possible output:
#     0.44989440807025205
#     0.7719221475578524
#     0.8117074999965667
#     0.3216176959290542
#     0.8296494377175666
    
# Choose a random item from a list.

fruits = ["apple", "banana", "cherry", "date", "elderberry", "fig", "grape"]
for i in range(5):
    print(random.choice(fruits))

# Possible output:
#     date
#     elderberry
#     apple
#     elderberry
#     grape
    
# Shuffle a list (in place).

vegetables = ["artichoke", "beet", "carrot", "daikon", "eggplant", "fennel"]
print("Before:", vegetables)
random.shuffle(vegetables)
print("After: ", vegetables)

# Possible output:
#     Before: ['artichoke', 'beet', 'carrot', 'daikon', 'eggplant', 'fennel']
#     After:  ['eggplant', 'carrot', 'artichoke', 'beet', 'daikon', 'fennel']
